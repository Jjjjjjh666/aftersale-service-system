    private int errNo;
    private String message;
    private int httpStatus = 200;

    private static final Map<Integer, ReturnNoBaK> returnNoMap;
    static {
        returnNoMap = new HashMap<>();
        for (ReturnNoBaK enum1 : values()) {
            returnNoMap.put(enum1.errNo, enum1);
        }
    }

    ReturnNoBaK(int code, String message, int httpStatus ){
        this.errNo = code;
        this.message = message;
        this.httpStatus = httpStatus;
    }

    ReturnNoBaK(int code, String message){
        this.errNo = code;
        this.message = message;
        this.httpStatus = 200;
    }

    public static ReturnNoBaK getReturnNoByCode(int code){
        return returnNoMap.get(code);
    }

    public int getErrNo() {
        return errNo;
    }

    public String getMessage(){
        return message;
    }

    public int getHttpStatus() {return httpStatus;}
